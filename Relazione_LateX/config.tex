% Pacchetto per la gestione e formattazione del codice sorgente
\usepackage{listings}
\usepackage{xcolor}

% Configurazione personalizzata di listings per codice e file di configurazione
\lstset{
    basicstyle=\ttfamily\small,        % Font monospaziato per il codice
    backgroundcolor=\color{gray!5},    % Sfondo leggermente grigio
    frame=single,                      % Bordo attorno al codice
    keywordstyle=\color{blue},         % Colore delle parole chiave
    commentstyle=\color{teal},         % Colore dei commenti
    stringstyle=\color{violet},        % Colore delle stringhe
    breaklines=true,                   % Va a capo automaticamente a fine riga
    showstringspaces=false,            % Non mostra gli spazi nelle stringhe
    keepspaces=true,                   % Mantiene la formattazione degli spazi
    numbers=left,                      % Numeri di riga a sinistra
    numberstyle=\tiny\color{gray},     % Stile dei numeri di riga
    captionpos=b                       % Didascalia posizionata in basso
}
